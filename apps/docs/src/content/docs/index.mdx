import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="Setup with Docker" icon="setting">
    ```bash
    docker run -d -p 3124:3124 --restart=always --name notymail \
      -e API_KEYS='' \
      -e MAIL_SENDER='notymail <test@1.1.1.1>' \
      -e MAIL_HOST='gmail.com' \
      -e MAIL_PORT='587' \
      -e MAIL_SECURE='false' \
      -e MAIL_AUTH_USER='test@1.1.1.1' \
      -e MAIL_AUTH_PASSWORD='password' \
      dafnik/notymail:latest
    ```
  </Card>
  <Card title="Use with curl" icon="pencil">
    ```bash
    curl -X 'POST' 'notymail.yourdomain.your_tld/api/email' \
      -H 'accept: application/json' \
      -H 'Content-Type: application/json' \
      -d '{
        "to": "test@1.1.1.1.at",
        "subject": "Hello world!",
        "body": "Hello body world! **This is markdown**"
      }'
    ```
  </Card>
  <Card title="Read the docs" icon="open-book">
    Learn more in [the notymail Docs](/getting-started/installation/).
